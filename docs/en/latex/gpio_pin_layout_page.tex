\hypertarget{gpio_pin_layout_page_autotoc_md12}{}\doxysubsubsection{\texorpdfstring{Notes}{Notes}}\label{gpio_pin_layout_page_autotoc_md12}

\begin{DoxyItemize}
\item Pins marked as "{}\+Free"{} are currently unused.
\item Strapping pins are reserved for boot/configuration and must not be modified at runtime.
\item Some GPIOs require pull-\/up or pull-\/down resistors as per ESP32-\/\+S3 guidelines.
\item All inputs use internal pull-\/ups.
\item All outputs are active-\/high and drive low-\/side loads via IRLZ44N MOSFETs. 
\end{DoxyItemize}
